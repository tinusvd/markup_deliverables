R Code for Price Generation

# Generate data for years 2006 to 2025
set.seed(42)
years <- 2006:2025
base_price <- 15 
annual_growth <- 0.05 

# Generate prices with random fluctuation
prices <- sapply(0:(length(years) - 1), function(i) {
  round(base_price * ((1 + annual_growth)^i) * runif(1, 0.95, 1.05), 2)
})
